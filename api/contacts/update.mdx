---
title: 'Update Contact'
api: 'PUT /api/contacts/:id'
description: 'Update an existing contact'
---

## Authentication

This endpoint requires JWT authentication.

<ParamField header="Authorization" type="string" required>
  Bearer token for authentication
</ParamField>

## Path Parameters

<ParamField path="id" type="number" required>
  The unique identifier of the contact to update
</ParamField>

## Body Parameters

<ParamField body="name" type="string">
  Name of the contact (max 128 characters)
</ParamField>

<ParamField body="email" type="string">
  Email address of the contact
</ParamField>

<ParamField body="number" type="string">
  Phone number of the contact
</ParamField>

<ParamField body="whatsapp" type="string">
  WhatsApp number of the contact
</ParamField>

## Response

<ResponseField name="id" type="number">
  Unique identifier for the contact
</ResponseField>

<ResponseField name="name" type="string">
  Updated name of the contact
</ResponseField>

<ResponseField name="email" type="string">
  Updated email address of the contact
</ResponseField>

<ResponseField name="number" type="string">
  Updated phone number of the contact
</ResponseField>

<ResponseField name="whatsapp" type="string">
  Updated WhatsApp number of the contact
</ResponseField>

<ResponseField name="ClientId" type="number">
  Foreign key reference to the associated client
</ResponseField>

<ResponseField name="createdAt" type="string">
  Timestamp when the contact was created
</ResponseField>

<ResponseField name="updatedAt" type="string">
  Timestamp when the contact was last updated
</ResponseField>

<RequestExample>
```bash cURL
curl --request PUT \
  --url https://api.devefficiency.com/api/contacts/1 \
  --header 'Authorization: Bearer YOUR_JWT_TOKEN' \
  --header 'Content-Type: application/json' \
  --data '{
    "name": "John Doe Updated",
    "email": "john.doe.new@example.com",
    "number": "+1234567899",
    "whatsapp": "+1234567899"
  }'
```
</RequestExample>
